bool LinkedList::del(char c) {
	for (Node *prev = head, *node; prev->next != nullptr; prev = node) {
		node = prev->next;
		if (node->data == c) {
			prev->next = node->next;
			delete node;
			return true;
		}
	}
	return false;
}
